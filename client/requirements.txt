#    pip install /home/mafiq/onnxruntime_gpu-1.18.0-cp310-cp310-linux_aarch64.whl
#
# 2. cuDNN 8 libraries are at: /home/mafiq/cudnn8/cudnn-linux-aarch64-8.9.5.30_cuda12-archive/lib
#    Set LD_LIBRARY_PATH before running:
#    export LD_LIBRARY_PATH=/home/mafiq/cudnn8/cudnn-linux-aarch64-8.9.5.30_cuda12-archive/lib:$LD_LIBRARY_PATH
#
# Or use the wrapper script: ./run_gpu.sh --debug
